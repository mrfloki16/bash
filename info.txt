######## Shell Scripting #######

# Wich shell?
$ echo $0
$ cat /etc/shells
$ cat /etc/passwd

# Enter in a specific shell
$ sh
$ csh
$ ksh

# Run a Script
## Absoluth path 
$ /home/mr_floki/script
## Relative path 
$ ./script

# Script naming convention
## Scripts Directory e.g.
$ /home/mr_floki/scripts
## Script name should identify function e.g.
$ /home/mr_floki/scripts/hello_world
## Script should end with .shell if multiple shells scripts e.g.
$ hello_world.bash

# Script file persmission
## Should have executable file permission
-rwxr-xr--
$  chmod +x file
$ chmod u=rwx file

#####CRONTAB############
# Make a script on boot:
$ crontab -e
# Min  Hour Day  Mon  Weekday
$ 0 * * * * 			# every hour
$ */15 * * * * 			# every 15 mins
$ 0 */2 * * * 			# every 2 hours
$ 0 0 * * 0 			# every Sunday midnight
$ @reboot 			# every reboot
# Adding tasks easily
$ echo "@reboot echo hi" | crontab
# Open in editor
$ crontab -e
# List tasks
$ crontab -l [-u user]
